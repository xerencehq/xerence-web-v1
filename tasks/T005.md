```typescript
'use client';

import React from 'react';
import Link from 'next/link';
import { motion } from 'framer-motion';
import { fadeInUpVariant, staggerContainerVariant, staggerItemVariant } from '@/styles';
import {
  FOOTER_LINKS,
  SOCIAL_LINKS,
  CONTACT_INFO,
  COMPANY_INFO,
} from './constants';
import {
  FooterWrapper,
  FooterContainer,
  FooterTop,
  FooterBrand,
  BrandLogo,
  BrandTagline,
  SocialLinks,
  SocialLink,
  FooterColumns,
  FooterColumn,
  ColumnTitle,
  ColumnList,
  ColumnLink,
  ContactItem,
  FooterCTA,
  CTATitle,
  CTAButton,
  FooterBottom,
  Copyright,
  LegalLinks,
  LegalLink,
} from './styles';

// Simple SVG icons
const SocialIcon: React.FC<{ icon: string }> = ({ icon }) => {
  switch (icon) {
    case 'twitter':
      return (
        <svg width="20" height="20" viewBox="0 0 24 24" fill="currentColor">
          <path d="M18.244 2.25h3.308l-7.227 8.26 8.502 11.24H16.17l-5.214-6.817L4.99 21.75H1.68l7.73-8.835L1.254 2.25H8.08l4.713 6.231zm-1.161 17.52h1.833L7.084 4.126H5.117z" />
        </svg>
      );
    case 'linkedin':
      return (
        <svg width="20" height="20" viewBox="0 0 24 24" fill="currentColor">
          <path d="M20.447 20.452h-3.554v-5.569c0-1.328-.027-3.037-1.852-3.037-1.853 0-2.136 1.445-2.136 2.939v5.667H9.351V9h3.414v1.561h.046c.477-.9 1.637-1.85 3.37-1.85 3.601 0 4.267 2.37 4.267 5.455v6.286zM5.337 7.433c-1.144 0-2.063-.926-2.063-2.065 0-1.138.92-2.063 2.063-2.063 1.14 0 2.064.925 2.064 2.063 0 1.139-.925 2.065-2.064 2.065zm1.782 13.019H3.555V9h3.564v11.452zM22.225 0H1.771C.792 0 0 .774 0 1.729v20.542C0 23.227.792 24 1.771 24h20.451C23.2 24 24 23.227 24 22.271V1.729C24 .774 23.2 0 22.222 0h.003z" />
        </svg>
      );
    case 'github':
      return (
        <svg width="20" height="20" viewBox="0 0 24 24" fill="currentColor">
          <path d="M12 0c-6.626 0-12 5.373-12 12 0 5.302 3.438 9.8 8.207 11.387.599.111.793-.261.793-.577v-2.234c-3.338.726-4.033-1.416-4.033-1.416-.546-1.387-1.333-1.756-1.333-1.756-1.089-.745.083-.729.083-.729 1.205.084 1.839 1.237 1.839 1.237 1.07 1.834 2.807 1.304 3.492.997.107-.775.418-1.305.762-1.604-2.665-.305-5.467-1.334-5.467-5.931 0-1.311.469-2.381 1.236-3.221-.124-.303-.535-1.524.117-3.176 0 0 1.008-.322 3.301 1.23.957-.266 1.983-.399 3.003-.404 1.02.005 2.047.138 3.006.404 2.291-1.552 3.297-1.23 3.297-1.23.653 1.653.242 2.874.118 3.176.77.84 1.235 1.911 1.235 3.221 0 4.609-2.807 5.624-5.479 5.921.43.372.823 1.102.823 2.222v3.293c0 .319.192.694.801.576 4.765-1.589 8.199-6.086 8.199-11.386 0-6.627-5.373-12-12-12z" />
        </svg>
      );
    default:
      return null;
  }
};

const Footer: React.FC = () => {
  return (
    <FooterWrapper data-testid="footer">
      <FooterContainer>
        <FooterTop
          as={motion.div}
          variants={staggerContainerVariant}
          initial="hidden"
          whileInView="visible"
          viewport={{ once: true, margin: '-100px' }}
        >
          {/* Brand Column */}
          <FooterBrand as={motion.div} variants={staggerItemVariant}>
            <BrandLogo href="/">Xerence</BrandLogo>
            <BrandTagline>{COMPANY_INFO.tagline}</BrandTagline>
            <SocialLinks>
              {SOCIAL_LINKS.map((social) => (
                <SocialLink
                  key={social.name}
                  href={social.href}
                  target="_blank"
                  rel="noopener noreferrer"
                  aria-label={social.name}
                >
                  <SocialIcon icon={social.icon} />
                </SocialLink>
              ))}
            </SocialLinks>
          </FooterBrand>

          {/* Link Columns */}
          <FooterColumns>
            <FooterColumn as={motion.div} variants={staggerItemVariant}>
              <ColumnTitle>Company</ColumnTitle>
              <ColumnList>
                {FOOTER_LINKS.company.map((link) => (
                  <li key={link.href}>
                    <ColumnLink href={link.href} as={Link}>
                      {link.label}
                    </ColumnLink>
                  </li>
                ))}
              </ColumnList>
            </FooterColumn>

            <FooterColumn as={motion.div} variants={staggerItemVariant}>
              <ColumnTitle>Services</ColumnTitle>
              <ColumnList>
                {FOOTER_LINKS.services.map((link) => (
                  <li key={link.href}>
                    <ColumnLink href={link.href} as={Link}>
                      {link.label}
                    </ColumnLink>
                  </li>
                ))}
              </ColumnList>
            </FooterColumn>

            <FooterColumn as={motion.div} variants={staggerItemVariant}>
              <ColumnTitle>Contact</ColumnTitle>
              <ColumnList>
                <ContactItem>
                  <a href={`mailto:${CONTACT_INFO.email}`}>
                    {CONTACT_INFO.email}
                  </a>
                </ContactItem>
                <ContactItem>{CONTACT_INFO.phone}</ContactItem>
                <ContactItem>{CONTACT_INFO.address}</ContactItem>
              </ColumnList>
            </FooterColumn>
          </FooterColumns>
        </FooterTop>

        {/* CTA Section */}
        <FooterCTA
          as={motion.div}
          variants={fadeInUpVariant}
          initial="hidden"
          whileInView="visible"
          viewport={{ once: true }}
        >
          <CTATitle>Ready to Start Your Project?</CTATitle>
          <CTAButton href="/book-consultation" as={Link}>
            Book a Free Consultation
          </CTAButton>
        </FooterCTA>

        {/* Bottom Bar */}
        <FooterBottom>
          <Copyright>
            © {COMPANY_INFO.year} {COMPANY_INFO.name}. All rights reserved.
          </Copyright>
          <LegalLinks>
            <LegalLink href="/privacy" as={Link}>
              Privacy Policy
            </LegalLink>
            <LegalLink href="/terms" as={Link}>
              Terms of Service
            </LegalLink>
          </LegalLinks>
        </FooterBottom>
      </FooterContainer>
    </FooterWrapper>
  );
};

export default Footer;
```

#### `src/components/Layout/Footer/Footer.test.tsx`
```typescript
import React from 'react';
import { render, screen } from '@/__tests__/test-utils';
import Footer from './index';
import { FOOTER_LINKS, SOCIAL_LINKS, COMPANY_INFO } from './constants';

describe('Footer', () => {
  it('should render the footer', () => {
    render(<Footer />);
    expect(screen.getByTestId('footer')).toBeInTheDocument();
  });

  it('should render the brand logo', () => {
    render(<Footer />);
    expect(screen.getByText('Xerence')).toBeInTheDocument();
  });

  it('should render the tagline', () => {
    render(<Footer />);
    expect(screen.getByText(COMPANY_INFO.tagline)).toBeInTheDocument();
  });

  it('should render social links', () => {
    render(<Footer />);
    SOCIAL_LINKS.forEach((social) => {
      expect(screen.getByLabelText(social.name)).toBeInTheDocument();
    });
  });

  it('should render company links', () => {
    render(<Footer />);
    FOOTER_LINKS.company.forEach((link) => {
      expect(screen.getByText(link.label)).toBeInTheDocument();
    });
  });

  it('should render services links', () => {
    render(<Footer />);
    FOOTER_LINKS.services.forEach((link) => {
      expect(screen.getByText(link.label)).toBeInTheDocument();
    });
  });

  it('should render CTA section', () => {
    render(<Footer />);
    expect(screen.getByText('Ready to Start Your Project?')).toBeInTheDocument();
    expect(screen.getByText('Book a Free Consultation')).toBeInTheDocument();
  });

  it('should render copyright with current year', () => {
    render(<Footer />);
    expect(screen.getByText(/Xerence Innovations, Ltd/)).toBeInTheDocument();
    expect(screen.getByText(new RegExp(String(COMPANY_INFO.year)))).toBeInTheDocument();
  });

  it('should render legal links', () => {
    render(<Footer />);
    expect(screen.getByText('Privacy Policy')).toBeInTheDocument();
    expect(screen.getByText('Terms of Service')).toBeInTheDocument();
  });

  it('should have correct link destinations', () => {
    render(<Footer />);
    const ctaLink = screen.getByText('Book a Free Consultation');
    expect(ctaLink).toHaveAttribute('href', '/book-consultation');
  });
});
```
