## Page Component

#### `src/app/about/page.tsx`
```typescript
import { Metadata } from 'next';
import Header from '@/components/Layout/Header';
import Footer from '@/components/Layout/Footer';
import { Section, H1, LeadText } from '@/components/Common';
import { Story } from '@/components/UI/AboutPage/Story';
import { Values } from '@/components/UI/AboutPage/Values';
import { Team } from '@/components/UI/AboutPage/Team';
import { Milestones } from '@/components/UI/AboutPage/Milestones';
import { CTASection } from '@/components/UI/CTASection';
import { TEAM_MEMBERS, SHOW_TEAM_SECTION } from '@/data/team';
import { MILESTONES } from '@/data/milestones';

export const metadata: Metadata = {
  title: 'About Us | Xerence Innovations',
  description: 'Learn about Xerence Innovations - our story, values, team, and mission to build intelligent software solutions.',
};

export default function AboutPage() {
  return (
    <>
      <Header />
      <main>
        <Section padding="xl" centered>
          <H1>About Xerence</H1>
          <LeadText>
            Building tomorrow's technology, today.
          </LeadText>
        </Section>

        <Story />

        <Values />

        {SHOW_TEAM_SECTION && <Team members={TEAM_MEMBERS} />}

        <Milestones milestones={MILESTONES} />

        <CTASection
          title="Want to work with us?"
          subtitle="Let's discuss how we can help bring your vision to life."
          ctaText="Book a Free Consultation"
          ctaHref="/book-consultation"
        />
      </main>
      <Footer />
    </>
  );
}
```
